#include <reg51.h>
#include <intrins.h>

unsigned char n;
sbit led1 = P2^0;
void Delay10ms(unsigned int c);

void main(void)
{
    led1 = 0; // 给定一个初始值
    while(1)
    {
        for (n=0; n<7; n++)
        {
            // 灯泡从右到左
            P2 = _crol_(P2, 1);
            Delay10ms(10);
        }
        for (n=0; n<7; n++)
        {
            // 灯泡从左到右
            P2 = _cror_(P2, 1);
            Delay10ms(10); // 延迟 0.1 秒
        }
    }
}

void Delay10ms(unsigned int c)
{
    unsigned char a,b;
    for(;c>0;c--)
        for(b=38;b>0;b--)
            for(a=130;a>0;a--);
}
